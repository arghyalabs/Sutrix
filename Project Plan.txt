# utils/file_handler.py
from pathlib import Path
from typing import Dict, List, Optional, Tuple
import pandas as pd
import chardet
from dataclasses import dataclass

@dataclass
class IngestionResult:
    """Result of file ingestion operation."""
    success: bool
    file_path: str
    sheet_names: List[str]
    dataframes: Dict[str, pd.DataFrame]
    encoding: str
    errors: List[str]
    metadata: dict

class FileIngestionEngine:
    """Handles CSV and XLSX file ingestion with robust error handling."""
    
    SUPPORTED_EXTENSIONS = {'.csv', '.xlsx', '.xlsm', '.xls'}
    MAX_FILE_SIZE_MB = 100
    
    def __init__(self, upload_dir: str = "uploads"):
        self.upload_dir = Path(upload_dir)
        self.upload_dir.mkdir(parents=True, exist_ok=True)
    
    def detect_encoding(self, file_path: str) -> str:
        """Detect file encoding using chardet."""
        with open(file_path, 'rb') as f:
            raw_data = f.read(10000)
            result = chardet.detect(raw_data)
            return result.get('encoding', 'utf-8')
    
    def validate_file(self, file_path: str) -> Tuple[bool, List[str]]:
        """Validate file structure and return errors."""
        errors = []
        path = Path(file_path)
        
        # Check extension
        if path.suffix.lower() not in self.SUPPORTED_EXTENSIONS:
            errors.append(f"Unsupported file type: {path.suffix}")
        
        # Check file size
        size_mb = path.stat().st_size / (1024 * 1024)
        if size_mb > self.MAX_FILE_SIZE_MB:
            errors.append(f"File too large: {size_mb:.1f}MB (max: {self.MAX_FILE_SIZE_MB}MB)")
        
        # Check if file is readable
        try:
            if path.suffix.lower() == '.csv':
                encoding = self.detect_encoding(file_path)
                pd.read_csv(file_path, encoding=encoding, nrows=1)
            else:
                pd.read_excel(file_path, nrows=1)
        except Exception as e:
            errors.append(f"Cannot read file: {str(e)}")
        
        return len(errors) == 0, errors
    
    def ingest_csv(self, file_path: str, **kwargs) -> Dict[str, pd.DataFrame]:
        """Ingest CSV file with encoding detection."""
        encoding = kwargs.get('encoding') or self.detect_encoding(file_path)
        
        # Default parameters for large file handling
        default_kwargs = {
            'encoding': encoding,
            'low_memory': False,
            'dtype_backend': 'numpy_nullable',
        }
        default_kwargs.update(kwargs)
        
        df = pd.read_csv(file_path, **default_kwargs)
        return {'Sheet1': df}
    
    def ingest_excel(self, file_path: str, **kwargs) -> Dict[str, pd.DataFrame]:
        """Ingest Excel file with multiple sheet support."""
        xl = pd.ExcelFile(file_path)
        dataframes = {}
        
        for sheet_name in xl.sheet_names:
            df = pd.read_excel(xl, sheet_name=sheet_name, **kwargs)
            dataframes[sheet_name] = df
        
        return dataframes
    
    def ingest(self, file_path: str, **kwargs) -> IngestionResult:
        """Main ingestion method."""
        errors = []
        path = Path(file_path)
        
        # Validate file
        is_valid, validation_errors = self.validate_file(file_path)
        if not is_valid:
            return IngestionResult(
                success=False,
                file_path=file_path,
                sheet_names=[],
                dataframes={},
                encoding='',
                errors=validation_errors,
                metadata={}
            )
        
        try:
            if path.suffix.lower() == '.csv':
                dataframes = self.ingest_csv(file_path, **kwargs)
            else:
                dataframes = self.ingest_excel(file_path, **kwargs)
            
            return IngestionResult(
                success=True,
                file_path=file_path,
                sheet_names=list(dataframes.keys()),
                dataframes=dataframes,
                encoding=self.detect_encoding(file_path),
                errors=[],
                metadata={
                    'total_rows': sum(df.shape[0] for df in dataframes.values()),
                    'total_cols': sum(df.shape[1] for df in dataframes.values()),
                    'file_size_bytes': path.stat().st_size
                }
            )
        except Exception as e:
            errors.append(f"Ingestion failed: {str(e)}")
            return IngestionResult(
                success=False,
                file_path=file_path,
                sheet_names=[],
                dataframes={},
                encoding='',
                errors=errors,
                metadata={}
            )

Error Handling Strategy

        'hepatotoxicity': ['hepatotoxicity', 'liver toxicity', 'hepatic',
                          'liver damage', 'hepatocellular'],
        'nephrotoxicity': ['nephrotoxicity', 'kidney toxicity', 'renal',
                          'kidney damage', 'nephropathy'],
        'neurotoxicity': ['neurotoxicity', 'neurological toxicity', 'neural',
                         'nervous system', 'neurobehavioral'],
        'reproductive_toxicity': ['reproductive_toxicity', 'reproductive toxicity',
                                 'reprotoxic', 'developmental toxicity',
                                 'teratogenicity', 'fertility']
    }
    
    def __init__(self):
        self._build_synonym_index()
    
    def _build_synonym_index(self) -> None:
        """Build reverse index for fast synonym lookup."""
        self._synonym_to_variable = {}
        
        all_synonyms = [
            self.SPECIES_SYNONYMS,
            self.ENDPOINT_SYNONYMS,
            self.CHEMICAL_IDENTIFIER_SYNONYMS,
            self.QUALIFIER_SYNONYMS,
            self.TOXICITY_CATEGORY_SYNONYMS
        ]
        
        for synonym_dict in all_synonyms:
            for variable, synonyms in synonym_dict.items():
                for synonym in synonyms:
                    self._synonym_to_variable[synonym.lower()] = variable
    
    def normalize_column_name(self, column_name: str) -> str:
        """Normalize column name by removing special characters and lowercasing."""
        normalized = column_name.strip().lower()
        normalized = re.sub(r'[\s\-_]+', '_', normalized)
        normalized = re.sub(r'[^a-z0-9_]', '', normalized)
        return normalized
    
    def suggest_mapping(self, column_name: str) -> List[ColumnMapping]:
        """Suggest scientific variable mappings for a column."""
        normalized = self.normalize_column_name(column_name)
        suggestions = []
        
        # Exact match
        if normalized in self._synonym_to_variable:
            suggestions.append(ColumnMapping(
                user_column=column_name,
                scientific_variable=self._synonym_to_variable[normalized],
                confidence=1.0,
                method='exact'
            ))
        
        # Check all synonyms for partial matches
        for synonym, variable in self._synonym_to_variable.items():
            if normalized in synonym or synonym in normalized:
                if not any(s.scientific_variable == variable for s in suggestions):
                    suggestions.append(ColumnMapping(
                        user_column=column_name,
                        scientific_variable=variable,
                        confidence=0.8,
                        method='fuzzy'
                    ))
        
        return suggestions
    
    def get_all_scientific_variables(self) -> Dict[str, List[str]]:
        """Get all available scientific variables and their synonyms."""
        return {
            'species': self.SPECIES_SYNONYMS,
            'endpoints': self.ENDPOINT_SYNONYMS,
            'chemical_identifiers': self.CHEMICAL_IDENTIFIER_SYNONYMS,
            'qualifiers': self.QUALIFIER_SYNONYMS,
            'toxicity_categories': self.TOXICITY_CATEGORY_SYNONYMS
        }

Mapping Persistence Strategy

# utils/config.py
import json
from pathlib import Path
from typing import Dict, List, Optional
from datetime import datetime

class MappingPersistence:
    """Handles persistence of column mappings."""
    
    def __init__(self, storage_dir: str = "outputs/mappings"):
        self.storage_dir = Path(storage_dir)
        self.storage_dir.mkdir(parents=True, exist_ok=True)
    
    def save_mapping(self, dataset_id: str, mappings: List[dict]) -> str:
        """Save column mappings to JSON file."""
        mapping_doc = {
            "dataset_id": dataset_id,
            "mappings": mappings,
            "created_at": datetime.now().isoformat(),
            "version": "1.0"
        }
        
        file_path = self.storage_dir / f"{dataset_id}_mapping.json"
        with open(file_path, 'w') as f:
            json.dump(mapping_doc, f, indent=2)
        
        return str(file_path)
    
    def load_mapping(self, dataset_id: str) -> Optional[dict]:
        """Load column mappings from JSON file."""
        file_path = self.storage_dir / f"{dataset_id}_mapping.json"
        if file_path.exists():
            with open(file_path, 'r') as f:
                return json.load(f)
        return None
    
    def export_mapping_template(self, output_path: str) -> None:
        """Export a template for manual mapping."""
        template = {
            "dataset_id": "your_dataset_id",
            "mappings": [
                {
                    "user_column": "Column_A_From_Excel",
                    "scientific_variable": "chemical_id",
                    "confidence": 1.0,
                    "method": "manual"
                }
            ],
            "available_variables": [
                "chemical_id", "cas_number", "smiles", "species", 
                "endpoint", "duration", "value", "unit", "qualifier"
            ]
        }
        
        with open(output_path, 'w') as f:
            json.dump(template, f, indent=2)

class TemplateManager:
    """Manages scientific domain templates."""
    
    def __init__(self, templates_dir: str = "templates"):
        self.templates_dir = Path(templates_dir)
        self._templates: Dict[str, DomainTemplate] = {}
        self._load_templates()
    
    def _load_templates(self) -> None:
        """Load all templates from the templates directory."""
        for template_file in self.templates_dir.glob("*.json"):
            try:
                with open(template_file, 'r') as f:
                    data = json.load(f)
                    template = DomainTemplate(
                        template_id=data['template_id'],
                        name=data['name'],
                        description=data['description'],
                        version=data['version'],
                        required_variables=data['required_variables'],
                        optional_variables=data.get('optional_variables', []),
                        default_hierarchy=data['default_hierarchy'],
                        validation_rules=data.get('validation_rules', {}),
                        metadata_standards=data.get('metadata_standards', {})
                    )
                    self._templates[template.template_id] = template
            except (json.JSONDecodeError, KeyError) as e:
                print(f"Warning: Could not load template {template_file}: {e}")
    
    def get_template(self, template_id: str) -> Optional[DomainTemplate]:
        """Get a template by ID."""
        return self._templates.get(template_id)
    
    def list_templates(self) -> List[Dict[str, str]]:
        """List all available templates."""
        return [
            {
                'id': t.template_id,
                'name': t.name,
                'description': t.description,
                'version': t.version
            }
            for t in self._templates.values()
        ]
    
    def validate_dataset_against_template(self, columns: List[str], 
                                          template_id: str) -> Dict:
        """Validate dataset columns against template requirements."""
        template = self.get_template(template_id)
        if not template:
            return {'valid': False, 'error': 'Template not found'}
        
        columns_lower = [c.lower() for c in columns]
        missing_required = []
        matched_optional = []
        
        for req_var in template.required_variables:
            if not any(req_var.lower() in col for col in columns_lower):
                missing_required.append(req_var)
        
        for opt_var in template.optional_variables:
            if any(opt_var.lower() in col for col in columns_lower):
                matched_optional.append(opt_var)
        
        return {
            'valid': len(missing_required) == 0,
            'template': template.name,
            'missing_required': missing_required,
            'matched_optional': matched_optional,
            'coverage': len(matched_optional) / max(len(template.optional_variables), 1)
        }

# segregation/engine.py
from pathlib import Path
from typing import Dict, List, Optional, Any, Tuple
from dataclasses import dataclass, field
from collections import defaultdict
import pandas as pd
import json
from datetime import datetime

@dataclass
class SegregationNode:
    """Represents a node in the segregation hierarchy."""
    level: int
    variable: str
    value: str
    data: pd.DataFrame
    children: List['SegregationNode'] = field(default_factory=list)
    path: str = ""

@dataclass
class SegregationResult:
    """Result of segregation operation."""
    root_path: str
    total_folders: int
    total_files: int
    hierarchy_levels: int
    leaf_nodes: List[dict]
    statistics: dict

class HierarchicalSegregationEngine:
    """Core engine for hierarchical dataset segregation."""
    
    def __init__(self, output_dir: str = "outputs"):
        self.output_dir = Path(output_dir)
        self.output_dir.mkdir(parents=True, exist_ok=True)
    
    def build_hierarchy_tree(self, df: pd.DataFrame, 
                             hierarchy: List[str]) -> SegregationNode:
        """
        Recursively build a hierarchy tree from dataframe.
        
        Args:
            df: Input dataframe
            hierarchy: List of column names defining hierarchy order
            
        Returns:
            Root node of the hierarchy tree
        """
        if not hierarchy:
            return SegregationNode(
                level=0,
                variable="root",
                value="all",
                data=df
            )
        
        current_var = hierarchy[0]
        remaining_hierarchy = hierarchy[1:]
        
        root = SegregationNode(
            level=0,
            variable="root",
            value="all",
            data=df
        )
        
        # Group by current hierarchy level
        if current_var not in df.columns:
            # Skip missing hierarchy levels
            if remaining_hierarchy:
                return self.build_hierarchy_tree(df, remaining_hierarchy)
            return root
        
        grouped = df.groupby(current_var, dropna=False)
        
        for group_value, group_df in grouped:
            # Create sanitized folder name
            safe_value = self._sanitize_folder_name(str(group_value))
            path = f"{current_var}={safe_value}"
            
            child_node = SegregationNode(
                level=1,
                variable=current_var,
                value=str(group_value),
                data=group_df,
                path=path
            )
            
            # Recursively build children
            if remaining_hierarchy:
                child_tree = self.build_hierarchy_tree(group_df, remaining_hierarchy)
                child_tree.path = f"{path}/{child_tree.path}"
                child_node.children.append(child_tree)
            
            root.children.append(child_node)
        
        return root
    
    def _sanitize_folder_name(self, name: str) -> str:
        """Sanitize string for use as folder name."""
        # Remove or replace invalid characters
        invalid_chars = '<>:"/\\|?*'
        for char in invalid_chars:
            name = name.replace(char, '_')
        # Truncate long names
        if len(name) > 100:
            name = name[:100]
        return name.strip('. ')
    
    def generate_folders(self, root: SegregationNode, 
                         base_path: Path,
                         export_format: str = 'xlsx') -> Tuple[int, int, List[dict]]:
        """
        Recursively generate folder structure and export files.
        
        Args:
            root: Root node of hierarchy tree
            base_path: Base output directory
            export_format: Output file format (xlsx, csv)
            
        Returns:
            Tuple of (total_folders, total_files, leaf_node_info)
        """
        total_folders = 0
        total_files = 0
        leaf_nodes = []
        
        if not root.children:
            # Leaf node - export data
            if not root.data.empty:
                folder_path = base_path / root.path if root.path != "all" else base_path
                folder_path.mkdir(parents=True, exist_ok=True)
                
                # Generate filename
                timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
                if export_format == 'xlsx':
                    filename = f"data_{timestamp}.xlsx"
                else:
                    filename = f"data_{timestamp}.csv"
                
                file_path = folder_path / filename
                
                # Export data
                if export_format == 'xlsx':
                    root.data.to_excel(file_path, index=False)
                else:
                    root.data.to_csv(file_path, index=False)
                
                total_files += 1
                
                leaf_nodes.append({
                    'path': str(folder_path),
                    'filename': filename,
                    'records': len(root.data),
                    'compounds': root.data.iloc[:, 0].nunique() if len(root.data.columns) > 0 else 0
                })
        else:
            # Non-leaf node - create folder and process children
            folder_path = base_path / root.path if root.path != "all" else base_path
            folder_path.mkdir(parents=True, exist_ok=True)
            total_folders += 1
            
            for child in root.children:
                sub_folders, sub_files, sub_leaves = self.generate_folders(
                    child, base_path, export_format
                )
                total_folders += sub_folders
                total_files += sub_files
                leaf_nodes.extend(sub_leaves)
        
        return total_folders, total_files, leaf_nodes
    
    def segregate(self, df: pd.DataFrame, 
                  hierarchy: List[str],
                  output_dir: Optional[str] = None,
                  export_format: str = 'xlsx',
                  session_id: Optional[str] = None) -> SegregationResult:
        """
        Main segregation method.
        
        Args:
            df: Input dataframe
            hierarchy: List of column names for hierarchy
            output_dir: Output directory (default: outputs/segregation_{session_id})
            export_format: Output file format
            session_id: Optional session identifier
            
        Returns:
            SegregationResult with all metadata
        """
        if output_dir is None:
            session = session_id or datetime.now().strftime("%Y%m%d_%H%M%S")
            output_dir = str(self.output_dir / f"segregation_{session}")
        
        output_path = Path(output_dir)
        output_path.mkdir(parents=True, exist_ok=True)
        
        # Build hierarchy tree
        root = self.build_hierarchy_tree(df, hierarchy)
        
        # Generate folders and export files
        total_folders, total_files, leaf_nodes = self.generate_folders(
            root, output_path, export_format
        )
        
        # Calculate statistics
        statistics = {
            'input_records': len(df),
            'input_columns': len(df.columns),
            'hierarchy_levels': len(hierarchy),
            'hierarchy_variables': hierarchy,
            'output_folders': total_folders,
            'output_files': total_files,
            'leaf_nodes': len(leaf_nodes),
            'avg_records_per_file': len(df) / max(total_files, 1),
            'min_records_per_file': min((ln['records'] for ln in leaf_nodes), default=0),
            'max_records_per_file': max((ln['records'] for ln in leaf_nodes), default=0)
        }
        
        # Save segregation metadata
        metadata = {
            'session_id': session_id or datetime.now().strftime("%Y%m%d_%H%M%S"),
            'timestamp': datetime.now().isoformat(),
            'hierarchy': hierarchy,
            'statistics': statistics,
            'leaf_nodes': leaf_nodes
        }
        
        metadata_path = output_path / 'segregation_metadata.json'
        with open(metadata_path, 'w') as f:
            json.dump(metadata, f, indent=2)
        
        return SegregationResult(
            root_path=str(output_path),
            total_folders=total_folders,
            total_files=total_files,
            hierarchy_levels=len(hierarchy),
            leaf_nodes=leaf_nodes,
            statistics=statistics
        )

FUNCTION segregate(dataframe, hierarchy_levels):
    IF hierarchy_levels is empty:
        RETURN single node containing all data
    
    current_level = hierarchy_levels[0]
    remaining_levels = hierarchy_levels[1:]
    
    root = create_node("root", "all", dataframe)
    
    FOR each unique value in current_level column:
        subset = filter(dataframe, current_level == value)
        sanitized_value = sanitize(value)
        path = current_level + "=" + sanitized_value
        
        child_node = create_node(current_level, value, subset, path)
        
        IF remaining_levels is not empty:
            child_tree = segregate(subset, remaining_levels)
            child_node.children.append(child_tree)
        
        root.children.append(child_node)
    
    RETURN root

FUNCTION export_tree(node, base_path):

        folder = base_path / node.path
        create_folder(folder)
        export_dataframe(node.data, folder / "data.xlsx")
        RETURN (1 folder, 1 file, leaf_info)
    ELSE:
        # Non-leaf node
        folder = base_path / node.path
        create_folder(folder)

        all_leaves = []

            total_folders += folders

# utils/qualifier_parser.py
from typing import Tuple, Optional, List
from dataclasses import dataclass
from enum import Enum
import re

class QualifierType(Enum):
    EXACT = "="
    GREATER_THAN = ">"
    LESS_THAN = "<"
    GREATER_OR_EQUAL = ">="
    LESS_OR_EQUAL = "<="
    NOT_DETECTED = "ND"
    BELOW_LOD = "<LOD"
    ABOVE_LOQ = ">LOQ"

@dataclass
class QualifiedValue:
    """Represents a value with its qualifier."""
    value: float
    qualifier: QualifierType
    original_string: str
    is_censored: bool
    
    @property
    def qsar_ready(self) -> bool:
        """Whether this value is suitable for QSAR modeling."""
        return self.qualifier == QualifierType.EXACT and not self.is_censored

class QualifierParser:
    """Parses and normalizes scientific qualifiers."""
    
    QUALIFIER_PATTERNS = {
        QualifierType.EXACT: [r'^=\s*([\d.eE+-]+)$', r'^([\d.eE+-]+)$'],
        QualifierType.GREATER_THAN: [r'^>\s*([\d.eE+-]+)$'],
        QualifierType.LESS_THAN: [r'^<\s*([\d.eE+-]+)$'],
        QualifierType.GREATER_OR_EQUAL: [r'^>=\s*([\d.eE+-]+)$', r'^≥\s*([\d.eE+-]+)$'],
        QualifierType.LESS_OR_EQUAL: [r'^<=\s*([\d.eE+-]+)$', r'^≤\s*([\d.eE+-]+)$'],
        QualifierType.NOT_DETECTED: [r'^nd$', r'^n\.d\.$', r'^not detected$', r'^nondetected$'],
        QualifierType.BELOW_LOD: [r'^<lod$', r'^<lod$', r'^below lod$', r'^below detection$'],
        QualifierType.ABOVE_LOQ: [r'^>loq$', r'^>loq$', r'^above loq$', r'^above quantification$']
    }
    
    def parse(self, value_str: str, detection_limit: Optional[float] = None) -> Optional[QualifiedValue]:
        """Parse a qualified value string."""
        if not value_str or not value_str.strip():
            return None
        
        original = value_str.strip()
        value_str = original.lower()
        
        for qualifier_type, patterns in self.QUALIFIER_PATTERNS.items():
            for pattern in patterns:
                match = re.match(pattern, value_str, re.IGNORECASE)
                if match:
                    if qualifier_type in [QualifierType.NOT_DETECTED, 
                                         QualifierType.BELOW_LOD]:
                        # Censored value - use detection limit if available
                        value = detection_limit if detection_limit else 0.0
                        return QualifiedValue(
                            value=value,
                            qualifier=qualifier_type,
                            original_string=original,
                            is_censored=True
                        )
                    elif qualifier_type == QualifierType.ABOVE_LOQ:
                        # Above quantification - special handling
                        return QualifiedValue(
                            value=float('inf'),
                            qualifier=qualifier_type,
                            original_string=original,
                            is_censored=True
                        )
                    else:
                        try:
                            numeric_value = float(match.group(1))
                            return QualifiedValue(
                                value=numeric_value,
                                qualifier=qualifier_type,
                                original_string=original,
                                is_censored=False
                            )
                        except (ValueError, IndexError):
                            continue
        
        # Try parsing as plain number
        try:
            return QualifiedValue(
                value=float(value_str),
                qualifier=QualifierType.EXACT,
                original_string=original,
                is_censored=False
            )
        except ValueError:
            return None
    
    def parse_column(self, values: List[str], 
                     detection_limit: Optional[float] = None) -> List[Optional[QualifiedValue]]:
        """Parse a column of values."""
        return [self.parse(str(v), detection_limit) for v in values]
    
    def get_qsar_ready_mask(self, qualified_values: List[Optional[QualifiedValue]]) -> List[bool]:
        """Get boolean mask for QSAR-ready values."""
        return [qv is not None and qv.qsar_ready for qv in qualified_values]
    
    def separate_censored(self, qualified_values: List[Optional[QualifiedValue]]
                         ) -> Tuple[List[QualifiedValue], List[QualifiedValue]]:
        """Separate exact values from censored values."""
        exact = []
        censored = []
        
        for qv in qualified_values:
            if qv is None:
                continue
            if qv.is_censored:
                censored.append(qv)
            else:
                exact.append(qv)
        
        return exact, censored

from typing import Dict, Optional, Tuple
from dataclasses import dataclass

@dataclass
class UnitDefinition:
    """Definition of a unit with conversion factors."""
    name: str
    category: str  # mass, concentration, dose, etc.
    base_unit: str
    conversion_factor: float  # Factor to convert to base unit
    aliases: list

class UnitConverter:
    """Handles unit normalization and conversion."""
    
    UNIT_DEFINITIONS = {
        # Mass units (base: mg)
        'mg': UnitDefinition('mg', 'mass', 'mg', 1.0, ['milligram', 'milligrams']),
        'g': UnitDefinition('g', 'mass', 'mg', 1000.0, ['gram', 'grams']),
        'kg': UnitDefinition('kg', 'mass', 'mg', 1000000.0, ['kilogram', 'kilograms']),
        'ug': UnitDefinition('ug', 'mass', 'mg', 0.001, ['microgram', 'micrograms', 'μg']),
        
        # Concentration units (base: mg/L)
        'mg/L': UnitDefinition('mg/L', 'concentration', 'mg/L', 1.0, 
                              ['milligram per liter', 'mg/l']),
        'ug/L': UnitDefinition('ug/L', 'concentration', 'mg/L', 0.001,
                              ['microgram per liter', 'ug/l', 'μg/L']),
        'ppm': UnitDefinition('ppm', 'concentration', 'mg/L', 1.0,
                             ['parts per million', 'ppm']),  # For water
        'ppb': UnitDefinition('ppb', 'concentration', 'mg/L', 0.001,
                             ['parts per billion', 'ppb']),  # For water
        
        # Dose units (base: mg/kg)
        'mg/kg': UnitDefinition('mg/kg', 'dose', 'mg/kg', 1.0,
                               ['milligram per kilogram', 'milligram/kilogram']),
        'g/kg': UnitDefinition('g/kg', 'dose', 'mg/kg', 1000.0,
                              ['gram per kilogram', 'gram/kilogram']),
        'ug/kg': UnitDefinition('ug/kg', 'dose', 'mg/kg', 0.001,
                               ['microgram per kilogram', 'microgram/kilogram']),
        
        # Dose rate units (base: mg/kg/day)
        'mg/kg/day': UnitDefinition('mg/kg/day', 'dose_rate', 'mg/kg/day', 1.0,
                                   ['milligram per kilogram per day']),
        'mg/kg/d': UnitDefinition('mg/kg/d', 'dose_rate', 'mg/kg/day', 1.0,
                                 ['milligram per kilogram per day']),
    }
    
    def normalize_unit(self, unit_str: str) -> Optional[str]:
        """Normalize unit string to standard form."""
        normalized = unit_str.strip().lower().replace(' ', '')
        
        for key, definition in self.UNIT_DEFINITIONS.items():
            if normalized == key.lower():
                return key
            for alias in definition.aliases:
                if normalized == alias.lower().replace(' ', ''):
                    return key
        
        return None
    
    def convert(self, value: float, from_unit: str, to_unit: str) -> Optional[float]:
        """Convert value from one unit to another."""
        from_def = self.UNIT_DEFINITIONS.get(from_unit)
        to_def = self.UNIT_DEFINITIONS.get(to_unit)
        
        if not from_def or not to_def:
            return None
        
        if from_def.category != to_def.category:
            return None  # Incompatible units
        
        # Convert to base unit, then to target unit
        base_value = value * from_def.conversion_factor
        target_value = base_value / to_def.conversion_factor
        
        return target_value
    
    def validate_unit_consistency(self, values: list, units: list) -> Dict:
        """Validate unit consistency across a dataset."""
        normalized_units = [self.normalize_unit(u) for u in units]
        unique_units = set(u for u in normalized_units if u is not None)
        
        return {
            'is_consistent': len(unique_units) <= 1,
            'unique_units': list(unique_units),
            'unit_counts': {u: normalized_units.count(u) for u in unique_units},
            'missing_units': normalized_units.count(None)
        }

class MasterIndexGenerator:
    """Generates master index for segregated datasets."""
    
    def __init__(self, output_dir: str = "outputs"):
        self.output_dir = Path(output_dir)
    
    def generate_from_segregation_result(self, result, 
                                         segregation_metadata: dict) -> str:
        """Generate master index from segregation result."""
        entries = []
        
        for leaf in result.leaf_nodes:
            # Parse hierarchy path
            path_parts = leaf['path'].split('/')
            hierarchy_info = self._parse_hierarchy_path(path_parts)
            
            entry = IndexEntry(
                filename=leaf['filename'],
                hierarchy_path=leaf['path'],
                record_count=leaf['records'],
                compound_count=leaf['compounds'],
                endpoint=hierarchy_info.get('endpoint'),
                species=hierarchy_info.get('species'),
                duration=hierarchy_info.get('duration'),
                filters_applied=[f"{k}={v}" for k, v in hierarchy_info.items()],
                quality_notes=[],
                generated_at=datetime.now().isoformat()
            )
            entries.append(entry)
        
        # Create DataFrame
        df = pd.DataFrame([asdict(e) for e in entries])
        
        # Add summary statistics
        summary = {
            'total_files': len(entries),
            'total_records': sum(e.record_count for e in entries),
            'total_compounds': sum(e.compound_count for e in entries),
            'unique_endpoints': len(set(e.endpoint for e in entries if e.endpoint)),
            'unique_species': len(set(e.species for e in entries if e.species)),
            'generated_at': datetime.now().isoformat()
        }
        
        # Save index
        index_path = self.output_dir / 'master_index.xlsx'
        with pd.ExcelWriter(index_path, engine='openpyxl') as writer:
            df.to_excel(writer, sheet_name='File Index', index=False)
            
            # Add summary sheet
            summary_df = pd.DataFrame([summary])
            summary_df.to_excel(writer, sheet_name='Summary', index=False)

                hierarchy[key] = value
        return hierarchy
    
    def generate_json_index(self, entries: List[IndexEntry]) -> str:

        }
        
        index_path = self.output_dir / 'master_index.json'
        with open(index_path, 'w') as f:
            json.dump(index_data, f, indent=2)
        
        return str(index_path)

import pandas as pd
import numpy as np
from typing import Dict, List, Optional, Tuple
from dataclasses import dataclass, asdict
from pathlib import Path
import json

@dataclass
class AuditFinding:
    """Represents a single audit finding."""
    category: str
    severity: str  # 'critical', 'warning', 'info'
    description: str
    affected_records: int
    details: dict

@dataclass
class AuditReport:
    """Complete audit report."""
    dataset_id: str
    total_records: int
    total_columns: int
    findings: List[AuditFinding]
    quality_score: float
    summary: dict

class ScientificAuditor:
    """Performs comprehensive audit of scientific datasets."""
    
    def __init__(self):
        self.findings: List[AuditFinding] = []
    
    def audit(self, df: pd.DataFrame, 
              column_mappings: dict,
              dataset_id: str = "unknown") -> AuditReport:
        """Run complete audit on dataset."""
        self.findings = []
        
        # Run all audit checks
        self._check_missing_values(df)
        self._check_duplicates(df, column_mappings)
        self._check_unit_consistency(df, column_mappings)
        self._check_endpoint_consistency(df, column_mappings)
        self._check_value_ranges(df, column_mappings)
        self._check_data_types(df)
        
        # Calculate quality score
        quality_score = self._calculate_quality_score(df)
        
        # Generate summary
        summary = {
            'total_findings': len(self.findings),
            'critical_findings': len([f for f in self.findings if f.severity == 'critical']),
            'warning_findings': len([f for f in self.findings if f.severity == 'warning']),
            'info_findings': len([f for f in self.findings if f.severity == 'info'])
        }
        
        return AuditReport(
            dataset_id=dataset_id,
            total_records=len(df),
            total_columns=len(df.columns),
            findings=self.findings,
            quality_score=quality_score,
            summary=summary
        )
    
    def _check_missing_values(self, df: pd.DataFrame) -> None:
        """Check for missing values in each column."""
        missing_counts = df.isnull().sum()
        total = len(df)
        
        for col, count in missing_counts.items():
            pct = (count / total) * 100
            if pct > 0:
                severity = 'critical' if pct > 30 else 'warning' if pct > 10 else 'info'
                self.findings.append(AuditFinding(
                    category='missing_values',
                    severity=severity,
                    description=f"Column '{col}' has {count} missing values ({pct:.1f}%)",
                    affected_records=int(count),
                    details={'column': col, 'missing_count': int(count), 'percentage': round(pct, 2)}
                ))
    
    def _check_duplicates(self, df: pd.DataFrame, 
                         column_mappings: dict) -> None:
        """Check for exact and near duplicates."""
        # Exact duplicates
        duplicate_mask = df.duplicated(keep=False)
        exact_duplicates = duplicate_mask.sum()
        
        if exact_duplicates > 0:
            self.findings.append(AuditFinding(
                category='duplicates',
                severity='warning',
                description=f"Found {exact_duplicates} exact duplicate rows",
                affected_records=int(exact_duplicates),
                details={'duplicate_count': int(exact_duplicates)}
            ))
        
        # Check for duplicate chemical identifiers
        if 'chemical_id' in column_mappings:
            col = column_mappings['chemical_id']
            if col in df.columns:
                dup_chemicals = df[col].duplicated(keep=False).sum()
                if dup_chemicals > 0:
                    self.findings.append(AuditFinding(
                        category='duplicate_chemicals',
                        severity='info',
                        description=f"Found {dup_chemicals} records with duplicate chemical identifiers",
                        affected_records=int(dup_chemicals),
                        details={'column': col}
                    ))
    
    def _check_unit_consistency(self, df: pd.DataFrame,
                                column_mappings: dict) -> None:
        """Check for unit consistency."""
        if 'unit' in column_mappings:
            col = column_mappings['unit']
            if col in df.columns:
                unique_units = df[col].nunique()
                if unique_units > 1:
                    unit_counts = df[col].value_counts().to_dict()
                    self.findings.append(AuditFinding(
                        category='unit_inconsistency',
                        severity='warning',
                        description=f"Found {unique_units} different units in column '{col}'",
                        affected_records=len(df),
                        details={'unique_units': unique_units, 'unit_distribution': unit_counts}
                    ))
    
    def _check_endpoint_consistency(self, df: pd.DataFrame,
                                    column_mappings: dict) -> None:
        """Check for endpoint consistency."""
        if 'endpoint' in column_mappings:
            col = column_mappings['endpoint']
            if col in df.columns:
                unique_endpoints = df[col].nunique()
                endpoint_counts = df[col].value_counts().to_dict()
                
                if unique_endpoints > 5:
                    self.findings.append(AuditFinding(
                        category='endpoint_variety',
                        severity='info',
                        description=f"Found {unique_endpoints} different endpoints",
                        affected_records=len(df),
                        details={'unique_endpoints': unique_endpoints, 
                                'endpoint_distribution': endpoint_counts}
                    ))
    
    def _check_value_ranges(self, df: pd.DataFrame,
                           column_mappings: dict) -> None:
        """Check for values outside expected ranges."""
        if 'value' in column_mappings:
            col = column_mappings['value']
            if col in df.columns:
                numeric_col = pd.to_numeric(df[col], errors='coerce')
                
                # Check for negative values (usually invalid for toxicity)
                negative_count = (numeric_col < 0).sum()
                if negative_count > 0:
                    self.findings.append(AuditFinding(
                        category='negative_values',
                        severity='warning',
                        description=f"Found {negative_count} negative values in '{col}'",
                        affected_records=int(negative_count),
                        details={'column': col}
                    ))
                
                # Check for extreme outliers
                if numeric_col.notna().any():
                    q1 = numeric_col.quantile(0.25)
                    q3 = numeric_col.quantile(0.75)
                    iqr = q3 - q1
                    upper_bound = q3 + 3 * iqr
                    
                    outlier_count = (numeric_col > upper_bound).sum()
                    if outlier_count > 0:
                        self.findings.append(AuditFinding(
                            category='outliers',
                            severity='info',
                            description=f"Found {outlier_count} potential outliers in '{col}'",
                            affected_records=int(outlier_count),
                            details={'column': col, 'upper_bound': float(upper_bound)}
                        ))
    
    def _check_data_types(self, df: pd.DataFrame) -> None:
        """Check for data type issues."""
        for col in df.columns:
            # Check for mixed types
            type_counts = df[col].apply(type).value_counts()
            if len(type_counts) > 1:
                self.findings.append(AuditFinding(
                    category='mixed_types',
                    severity='info',
                    description=f"Column '{col}' contains mixed data types",
                    affected_records=len(df),
                    details={'column': col, 'types': {str(k): int(v) for k, v in type_counts.items()}}
                ))
    
    def _calculate_quality_score(self, df: pd.DataFrame) -> float:
        """Calculate overall data quality score (0-100)."""
        score = 100.0
        
        # Deduct for missing values
        missing_pct = df.isnull().sum().sum() / (len(df) * len(df.columns)) * 100
        score -= missing_pct * 0.5
        
        # Deduct for duplicates
        dup_pct = df.duplicated().sum() / len(df) * 100
        score -= dup_pct * 0.3
        
        # Deduct for critical findings
        critical_count = len([f for f in self.findings if f.severity == 'critical'])
        score -= critical_count * 5
        
        # Deduct for warnings
        warning_count = len([f for f in self.findings if f.severity == 'warning'])
        score -= warning_count * 2
        
        return max(0, min(100, round(score, 1)))

from abc import ABC, abstractmethod
from typing import Any, Dict, List, Optional
from dataclasses import dataclass
import time

@dataclass
class PluginResult:
    """Result from a plugin operation."""
    success: bool
    data: Any
    error: Optional[str]
    cached: bool

class BaseAPIPlugin(ABC):
    """Base class for API plugins."""
    
    def __init__(self, cache_enabled: bool = True, 
                 rate_limit_delay: float = 0.1):
        self.cache_enabled = cache_enabled
        self.rate_limit_delay = rate_limit_delay
        self._cache: Dict[str, PluginResult] = {}
        self._last_request_time = 0
    
    def _respect_rate_limit(self) -> None:
        """Ensure rate limiting is respected."""
        elapsed = time.time() - self._last_request_time
        if elapsed < self.rate_limit_delay:
            time.sleep(self.rate_limit_delay - elapsed)
        self._last_request_time = time.time()
    
    def _get_cache_key(self, identifier: str) -> str:
        """Generate cache key for identifier."""
        return f"{self.__class__.__name__}:{identifier.lower().strip()}"
    
    def _get_cached(self, identifier: str) -> Optional[PluginResult]:
        """Get cached result if available."""
        if not self.cache_enabled:
            return None
        return self._cache.get(self._get_cache_key(identifier))
    
    def _cache_result(self, identifier: str, result: PluginResult) -> None:
        """Cache a result."""
        if self.cache_enabled:
            self._cache[self._get_cache_key(identifier)] = result
    
    @abstractmethod
    def fetch(self, identifier: str, **kwargs) -> PluginResult:
        """Fetch data for an identifier."""
        pass
    
    @abstractmethod
    def validate(self, identifier: str) -> bool:
        """Validate an identifier."""
        pass

import requests
from typing import Optional
from .base_plugin import BaseAPIPlugin, PluginResult

class PubChemPlugin(BaseAPIPlugin):
    """Plugin for PubChem API integration."""
    
    BASE_URL = "https://pubchem.ncbi.nlm.nih.gov/rest/pug"
    
    def fetch(self, identifier: str, 
              identifier_type: str = "name",
              properties: list = None,
              **kwargs) -> PluginResult:
        """Fetch compound data from PubChem."""
        # Check cache first
        cached = self._get_cached(identifier)
        if cached:
            return cached
        
        if properties is None:
            properties = ['MolecularWeight', 'MolecularFormula', 'CanonicalSMILES', 
                         'IsomericSMILES', 'InChI', 'InChIKey', 'IUPACName']
        
        try:
            self._respect_rate_limit()
            
            # Build URL
            props_str = ','.join(properties)
            url = f"{self.BASE_URL}/compound/{identifier_type}/{identifier}/property/{props_str}/json"
            
            response = requests.get(url, timeout=30)
            response.raise_for_status()
            
            data = response.json()
            
            if 'PropertyTable' in data and 'Properties' in data['PropertyTable']:
                result_data = data['PropertyTable']['Properties'][0]
                result = PluginResult(
                    success=True,
                    data=result_data,
                    error=None,
                    cached=False
                )
            else:
                result = PluginResult(
                    success=False,
                    data=None,
                    error="No properties found",
                    cached=False
                )
            
            self._cache_result(identifier, result)
            return result
            
        except requests.RequestException as e:
            return PluginResult(
                success=False,
                data=None,
                error=str(e),
                cached=False
            )
    
    def validate(self, identifier: str, 
                 identifier_type: str = "name") -> bool:
        """Validate a chemical identifier."""
        result = self.fetch(identifier, identifier_type)
        return result.success
    
    def get_smiles(self, identifier: str) -> Optional[str]:
        """Get SMILES string for a compound."""
        result = self.fetch(identifier, properties=['CanonicalSMILES'])
        if result.success:
            return result.data.get('CanonicalSMILES')
        return None
    
    def get_molecular_weight(self, identifier: str) -> Optional[float]:
        """Get molecular weight for a compound."""
        result = self.fetch(identifier, properties=['MolecularWeight'])
        if result.success:
            try:
                return float(result.data.get('MolecularWeight', 0))
            except (ValueError, TypeError):
                return None
        return None
